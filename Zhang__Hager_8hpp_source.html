49 {
50 optim_assert(arg.step > 0, "step must be positive.");
51 iter = 0; // reset iter
52 fp_t dTg; // d.dot(g) in sm prob or
53 // or (cur_x - prev_x) / step .dot(g_grad_map) in nsm prob
54 // begin line search
55 for (iter = 1; iter <= max_iter; iter++)
56 {
57 arg.step_forward(this->prob.get());
58 arg.update_cur_loss(this->prob.get());
59 if constexpr (use_prox)
60 {
61 arg.update_prev_grad_map(this->prob.get());
62 arg.tmp = arg.cur_x - arg.prev_x;
63 dTg = BMO_MAT_DOT_PROD(
64 arg.tmp, arg.prev_grad_map) /
65 arg.step;
66 }
67 else
68 dTg = BMO_MAT_DOT_PROD(arg.direction, arg.prev_grad);
69 optim_assert(dTg < 0, "dTg must be negtive.");
70 if (arg.cur_loss <= Cval + pho * arg.step * dTg ||
71 arg.step == this->min_step)
72 {
73 // update pQ, Q, Cval
74 pQ = Q, Q = gamma * Q + 1;
75 Cval = (gamma * pQ * Cval + arg.cur_loss) / Q;
76 goto over;
77 }
78 arg.step *= decay_rate;
79 // force step larger than min_step
80 if (arg.step < this->min_step)
81 arg.step = this->min_step;
82 }
83 // if line search failed, use the min step
84 // and give a warning(TODO)
85 arg.step = this->min_step;
86 arg.step_forward(this->prob.get());
87 arg.update_cur_loss(this->prob.get());
88 over:
89 if (this->update_cur_grad)
90 {
91 arg.update_cur_grad(this->prob.get());
92 if constexpr (use_prox)
93 arg.update_cur_grad_map(this->prob.get());
94 }
